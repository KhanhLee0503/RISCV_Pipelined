    ##############################################
    # Test R-Type Operations + Data Hazard (RAW)
    ##############################################

    # Base address for results
    lui     x10, 0x10000        # x10 = 0x1000_0000
    addi    x11, x10, 0         # x11 = write pointer

    # Initial operands
    addi    x5, x0, 10          # x5 = 10
    addi    x6, x0, 3           # x6 = 3

    ##############################################
    # 1. ADD
    ##############################################
    add     x7, x5, x6          # x7 = 10 + 3 = 13
    sw      x7, 0(x11)          # store result
    addi    x11, x11, 4         # next address

    ##############################################
    # 2. SUB (RAW hazard: uses x7)
    ##############################################
    sub     x7, x7, x6          # x7 = 13 - 3 = 10
    sw      x7, 0(x11)
    addi    x11, x11, 4

    ##############################################
    # 3. XOR (RAW)
    ##############################################
    xor     x7, x7, x6          # x7 = 10 xor 3 = 9
    sw      x7, 0(x11)
    addi    x11, x11, 4

    ##############################################
    # 4. OR (RAW)
    ##############################################
    or      x7, x7, x6          # x7 = 9 or 3 = 11
    sw      x7, 0(x11)
    addi    x11, x11, 4

    ##############################################
    # 5. AND (RAW)
    ##############################################
    and     x7, x7, x6          # x7 = 11 and 3 = 3
    sw      x7, 0(x11)
    addi    x11, x11, 4

    ##############################################
    # 6. SLL (RAW)
    ##############################################
    sll     x7, x7, x6          # x7 = 3 << 3 = 24
    sw      x7, 0(x11)
    addi    x11, x11, 4

    ##############################################
    # 7. SRL (RAW)
    ##############################################
    srl     x7, x7, x6          # x7 = 24 >> 3 = 3 (logical)
    sw      x7, 0(x11)
    addi    x11, x11, 4

    ##############################################
    # 8. SRA (RAW)
    ##############################################
    addi    x5, x0, -16         # x5 = 0xFFFFFFF0 (negative)
    sra     x7, x5, x6          # arithmetic shift: -16 >> 3 = -2
    sw      x7, 0(x11)
    addi    x11, x11, 4

    ##############################################
    # 9. SLT (signed)
    ##############################################
    addi    x5, x0, -5          # x5 = -5
    addi    x6, x0, 7           # x6 = 7
    slt     x7, x5, x6          # -5 < 7 → x7 = 1
    sw      x7, 0(x11)
    addi    x11, x11, 4

    ##############################################
    # 10. SLTU (unsigned)
    ##############################################
    sltu    x7, x5, x6          # (-5 unsigned) > 7 → x7 = 0
    sw      x7, 0(x11)
    addi    x11, x11, 4

    ##############################################
    # Stop Program
    ##############################################
halt:
    j halt
