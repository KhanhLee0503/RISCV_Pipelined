    .section .text
    .globl _start

_start:
    ################################################################
    # Chuẩn bị địa chỉ IO
    ################################################################
    lui     x10, 0x10000        # x10 = 0x1000_0000 (IO base)

    ################################################################
    # Bắt đầu test JAL liên tiếp
    ################################################################

    # Ghi số 1 → cho biết đang ở _start
    addi    x5, x0, 1
    sw      x5, 0(x10)

    ################################################################
    # JAL 1: nhảy tới label A
    ################################################################
    jal     x1, A               # x1 = return address sau JAL
    # Lệnh này bị skip khi nhảy
    addi    x5, x0, 99
    sw      x5, 0(x10)          # sẽ không chạy

A:
    # Ghi số 2 → cho biết đã đến A
    addi    x5, x0, 2
    sw      x5, 0(x10)

    ################################################################
    # JAL 2: nhảy tiếp tới label B
    ################################################################
    jal     x1, B
    addi    x5, x0, 98
    sw      x5, 0(x10)          # bị bỏ qua

B:
    # Ghi số 3 → cho biết đã đến B
    addi    x5, x0, 3
    sw      x5, 0(x10)

    ################################################################
    # JAL 3: nhảy tới C, kiểm tra stack JAL sâu hơn
    ################################################################
    jal     x1, C

C:
    # Ghi số 4 → đã tới C
    addi    x5, x0, 4
    sw      x5, 0(x10)

    ################################################################
    # Thử chuỗi JAL liên tiếp không delay
    ################################################################
    jal     x1, D       # nhảy ngay tiếp
    jal     x1, E       # nhảy chồng lên, test RA
    jal     x1, F       # nhảy sâu, kiểm JAL chain

D:
    addi    x5, x0, 5
    sw      x5, 0(x10)
    jalr x0, x1, 0

E:
    addi    x5, x0, 6
    sw      x5, 0(x10)
	jalr x0, x1, 0
F:
    addi    x5, x0, 7       # tới F
    sw      x5, 0(x10)

    ################################################################
    # Kết thúc test – loop forever
    ################################################################
END:
    j END
